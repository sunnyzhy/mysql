# 清理日志

日志文件存储目录：```/var/lib/mysql```

```bash
# 查看mysql目录
du -sh /var/lib/mysql/*

# 删除所有 binlog.数字 文件，保留 binlog.index
rm -f binlog.[0-9]* 

# 或者更精确地匹配（避免误删）
rm -f binlog.000*
```

## 1. 临时生效（立即执行，无需重启）

在 MySQL 命令行中执行以下命令，可立即开始自动清理：

```sql
-- 1. 检查当前 binlog 过期设置
SHOW VARIABLES LIKE 'binlog_expire_logs_seconds';
SHOW VARIABLES LIKE 'expire_logs_days';

-- 2. 如果存在 expire_logs_days 残留设置，先清空它（避免冲突）
SET GLOBAL expire_logs_days = 0;

-- 3. 设置 binlog 保留 7 天（604800 秒）
SET GLOBAL binlog_expire_logs_seconds = 604800;

-- 4. 验证设置是否生效
SHOW VARIABLES LIKE 'binlog_expire_logs_seconds';

-- 5. 手动触发清理（可选，MySQL 后台会自动清理，但手动执行可以立即生效）
-- 注意：FLUSH LOGS 不会立即删除旧日志，只是切换日志文件
-- 真正立即清理旧日志应该使用 PURGE 命令
PURGE BINARY LOGS BEFORE NOW() - INTERVAL 7 DAY;

-- 6. 查看当前 binlog 文件列表，确认清理结果
SHOW BINARY LOGS;
```

## 2. 永久生效（修改配置文件，需要重启）

编辑 MySQL 配置文件（`/etc/my.cnf` 或 `/etc/mysql/my.cnf`）：

```cnf
[mysqld]
# ============ Binlog 过期设置（二选一，推荐使用新参数）============
# 方式一：使用新参数（推荐，MySQL 8.0+）
binlog_expire_logs_seconds = 604800   # 7天 = 7*24*60*60

# 方式二：如果必须使用旧参数（不推荐，MySQL 8.0已弃用）
# expire_logs_days = 7                # 已弃用，与上面的冲突，请注释掉

# ============ 其他 Binlog 相关配置 ============
# 限制单个 binlog 文件大小（达到此值后自动切换）
max_binlog_size = 500M

# Binlog 文件命名前缀
log_bin = /var/lib/mysql/mysql-bin

# Binlog 格式（建议 ROW，便于主从复制和数据恢复）
binlog_format = ROW

# Binlog 缓存大小
binlog_cache_size = 32K

# 同步策略（重要！影响数据安全性和性能）
# 1 表示每次事务提交都同步磁盘（最安全，性能最差）
# 0 表示由操作系统决定何时同步（性能最好，但断电可能丢数据）
sync_binlog = 1
```

## 3. 编写自定义清理脚本 (Cron Job)

**示例脚本（clean_mysql_logs.sh）**：

```bash
#!/bin/bash
# ============================================
# MySQL Binlog & Slow Log 清理脚本
# 适用于 MySQL 8.0+
# ============================================

set -e  # 遇到错误立即退出

# ============ 配置区域 ============
MYSQL_USER="root"
MYSQL_PASS="YourPassword"  # 建议使用 .my.cnf 替代明文密码
MYSQL_HOST="localhost"
MYSQL_PORT="3306"
MYSQL_SOCKET="/var/lib/mysql/mysql.sock"

SLOW_LOG_DIR="/var/log/mysql"
SLOW_LOG_NAME="slow-queries.log"
SLOW_LOG_RETENTION_DAYS=3

BINLOG_RETENTION_DAYS=7

LOG_FILE="/var/log/mysql_cleanup.log"
MAX_LOG_SIZE=10485760  # 10MB，防止日志文件过大

# ============ 函数定义 ============

# 日志函数
log_message() {
    echo "[$(date '+%Y-%m-%d %H:%M:%S')] $1" | tee -a "$LOG_FILE"
}

# 日志轮转（防止日志文件无限增长）
rotate_log() {
    if [ -f "$LOG_FILE" ] && [ $(stat -c%s "$LOG_FILE") -gt $MAX_LOG_SIZE ]; then
        mv "$LOG_FILE" "${LOG_FILE}.old"
        touch "$LOG_FILE"
        log_message "日志文件已轮转（超过 ${MAX_LOG_SIZE} 字节）"
    fi
}

# ============ 主程序 ============

# 1. 初始化
rotate_log
log_message "========== 开始清理任务 =========="

# 2. 检查 MySQL 连接
log_message "检查 MySQL 连接..."
if ! mysqladmin -u"$MYSQL_USER" -p"$MYSQL_PASS" -h"$MYSQL_HOST" -P"$MYSQL_PORT" ping &>/dev/null; then
    log_message "错误：无法连接到 MySQL，请检查配置！"
    exit 1
fi
log_message "MySQL 连接正常"

# 3. 清理慢查询日志（安全方式）
log_message "开始清理慢查询日志（保留 ${SLOW_LOG_RETENTION_DAYS} 天）..."

# 方式一：只处理文件，保留目录结构
find "$SLOW_LOG_DIR" -name "$SLOW_LOG_NAME*" -mtime +${SLOW_LOG_RETENTION_DAYS} -type f | while read -r file; do
    log_message "  清空文件: $file"
    # 清空文件内容（保留文件本身）
    > "$file" 2>/dev/null || echo -n > "$file"
done

# 或者方式二：删除旧文件并重新创建（更彻底）
# find "$SLOW_LOG_DIR" -name "$SLOW_LOG_NAME*" -mtime +${SLOW_LOG_RETENTION_DAYS} -delete
# touch "$SLOW_LOG_DIR/$SLOW_LOG_NAME"
# log_message "已删除 ${SLOW_LOG_RETENTION_DAYS} 天前的慢查询日志"

# 4. 清理 Binlog（使用 MySQL 8.0 新参数方式）
log_message "开始清理 Binlog（保留 ${BINLOG_RETENTION_DAYS} 天）..."

# 获取当前 binlog 过期设置
CURRENT_EXPIRE=$(mysql -u"$MYSQL_USER" -p"$MYSQL_PASS" -h"$MYSQL_HOST" -P"$MYSQL_PORT" \
    -N -e "SELECT @@global.binlog_expire_logs_seconds;" 2>/dev/null)

if [ -z "$CURRENT_EXPIRE" ] || [ "$CURRENT_EXPIRE" = "NULL" ]; then
    log_message "警告：binlog_expire_logs_seconds 未设置，尝试使用 expire_logs_days"
    CURRENT_EXPIRE=$(mysql -u"$MYSQL_USER" -p"$MYSQL_PASS" -h"$MYSQL_HOST" -P"$MYSQL_PORT" \
        -N -e "SELECT @@global.expire_logs_days;" 2>/dev/null)
fi

log_message "当前 binlog 过期时间: ${CURRENT_EXPIRE} 秒 ($(($CURRENT_EXPIRE / 86400)) 天)"

# 执行清理（增加安全检查）
BINLOG_COUNT=$(mysql -u"$MYSQL_USER" -p"$MYSQL_PASS" -h"$MYSQL_HOST" -P"$MYSQL_PORT" \
    -N -e "SHOW BINARY LOGS;" | wc -l)

log_message "当前 binlog 文件数量: $BINLOG_COUNT"

if [ "$BINLOG_COUNT" -gt 2 ]; then
    # 使用 DATE_SUB 方式，精确到秒
    mysql -u"$MYSQL_USER" -p"$MYSQL_PASS" -h"$MYSQL_HOST" -P"$MYSQL_PORT" \
        -e "PURGE BINARY LOGS BEFORE DATE_SUB(NOW(), INTERVAL ${BINLOG_RETENTION_DAYS} DAY);" \
        2>&1 | tee -a "$LOG_FILE"
    
    if [ ${PIPESTATUS[0]} -eq 0 ]; then
        log_message "Binlog 清理成功"
    else
        log_message "错误：Binlog 清理失败"
        exit 1
    fi
else
    log_message "警告：binlog 文件数量过少（$BINLOG_COUNT），跳过清理"
fi

# 5. 清理后检查
REMAINING=$(mysql -u"$MYSQL_USER" -p"$MYSQL_PASS" -h"$MYSQL_HOST" -P"$MYSQL_PORT" \
    -N -e "SHOW BINARY LOGS;" | wc -l)

log_message "清理后 binlog 文件数量: $REMAINING"
log_message "========== 清理任务完成 =========="

exit 0
```

将此脚本添加到 Crontab 中即可实现定时执行：

```bash
# crontab -e
# 每天凌晨两点执行清理脚本
0 2 * * * /root/clean_mysql_logs.sh
```

自动清理策略设置好后，可以随时用 `SHOW VARIABLES LIKE 'expire_logs_days';` 或 `SHOW BINARY LOGS;` 来检查配置是否生效以及磁盘空间是否释放。
